\section{Potentialaufgabe}
Gegeben ist das Vektorfeld

\begin{equation*}
\vec{F}=\begin{pmatrix} -3x^2y^2z+y^3 \\ -2x^3yz+3xy^2-z \\ -x^3y^2-y\end{pmatrix}
\end{equation*}

\begin{enumerate}[label=\roman*]
\item {Bestimme $\mathrm{rot}\vec{F}$}
\item {Bestimme das Potential $\phi$ mit $\mathrm{grad}\phi=-\vec{F}$}
\end{enumerate}
